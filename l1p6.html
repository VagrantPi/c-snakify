分蘋果（Apple sharing）

有 N 個學生分 K 顆蘋果，平均分配，無法整除的部分留在籃子裡。先讀入 N（學生數）再讀入 K（蘋果數）。第一行印出每個學生分到幾顆，第二行印出籃子裡剩幾顆。

本地測試指令

① 用你的 IDE 把解答存成 solution.c。② 打開同一個資料夾的終端機。③ 複製下面整段貼上執行 —— 它會自動編譯並用上面所有測試項目檢查，回報通過/失敗。
cc solution.c -o sol || exit
p=0; f=0
t(){ g=$(printf '%s\n' "$1" | ./sol); if [ "$g" = "$2" ]; then echo "✓ $1"; p=$((p+1)); else printf '✗ 輸入[%s] 期望[%s] 得到[%s]\n' "$1" "$2" "$g"; f=$((f+1)); fi; }
t '3 10' $'3\n1'
t '5 17' $'3\n2'
t '4 4' $'1\n0'
t '2 7' $'3\n1'
t '6 20' $'3\n2'
echo "== 通過 $p / 失敗 $f =="
